def fibonacci(n):
    if n <= 0:
        return 0
    elif n == 1:
        return 1
    else:
        return fibonacci(n - 1) + fibonacci(n - 2)

# Function call
result = fibonacci(10)
print(f"Fibonacci of 10: {result}")

# Lists and loops
numbers = [1, 2, 3, 4, 5]
for num in numbers:
    print(f"Number: {num}")

# Conditionals
if result > 20:
    print("Result is greater than 20")
